༄༅། །དཔལ་ཕྱག་ན་རྡོ་རྗེ་ལ་ཕྱག་འཚལ་ལོ། །འཇིག་རྟེན་བི་ན་ཡ་ཀའི་གདོན་ལས་ཐར་བར་འདོད་པས། སྔོན་དུ་འཇུག་དུས་དང་མཚན་མ་ཤེས་པར་བྱ་སྟེ། དམངས་རིགས་ཀྱི་ལུས་ལས་དྲི་མ་འབྱུང་བའམ། གོས་ངན་གོན་པའི་ཚེའམ། ཁ་ཟས་གཞན་གྱི་ལྷག་མ་ཟ་
བའི་ཚེའམ། བུད་མེད་དང་དབྱེས་བྱས་པའི་ཚེའམ། བཤང་གཅིའི་དུས་སམ། གཅེར་བུར་གནས་པའི་དུས་སམ། མར་ངོའི་ཚེས་ལྔ་དང་བཅུ་བཞི་འམ། གནམ་སྟོང་ལ་འཇུག་གོ། །ཞུགས་པའི་མཚན་མ་ནི་མཁུན་ཞིང་ངུ་བ་དང་། ནོར་རྣམས་ཆུད་གསོན་དུ་འགྲོ་བ་དང་། གཞན་
གྱིས་ཡིད་དུ་མི་འོང་བ་དང་། སྐྱེས་པ་ཡིན་ན་བུད་མེད་མི་རྙེད་པ་དང་། བུད་མེད་ཡིན་ན་སྐྱེས་པ་མི་རྙེད་པ་དང་། ཚོང་བྱས་ན་འཕམ་པ་དང་། ཞིང་མི་སྐྱེ་ལ་གལ་ཏེ་བུད་མེད་རྙེད་ཀྱང་བུད་མེད་བསམ་པ་འགྱུར་བ་ནི་ལོག་འདྲེན་གྱི་གདོན་གྱིས་ཟིན་པ་སྟེ། དེ་ལས་ཐར་པའི་ཐབས་སྒྲུབ་པ་ནི།
གཏོར་མའི་བུམ་པ་བཞི་འབྲས་ཡོས་དང་། ཤ་བཙོས་དང་། རྗེན་པ་དང་། ཁུར་བ་དང་། དྲི་དང་། མེ་ཏོག་དཀར་པོའི་ཕྲེང་བ་དང་། འབྲས་ཐུག་དང་། སྤོས་སྣ་ཚོགས་དང་ལྡན་པའི་གཏོར་མ་ནུབ་ཏུ་སྦྱིན་ཏེ། དེ་ནས་གུ་གུལ་དང་། ཡུངས་ཀར་དང་། གི་ཝང་དང་། ཤུ་དག །རུ་རྟ།
སུག་སྨེལ། སྒྲོན་ཤིང་། ཐང་ཤིང་། ཙནྡན། ཨ་ཀ་རུ། ནཱ་ག་གེ་སར་རྣམས་ཆ་བསྙམས་ལ་བའི་ཆུ་དང་རའི་ཆུ་དང་བཏགས་ལ་སྔ་མ་བཞིན་དུ་མར་དུ་བཙོས་པ་ལ་སོགས་པ་བྱས་ལ། བྱུགས་པ་དང་བསྐུ་བ་ལ་སོགས་པ་བྱས་ན། ལོག་འདྲེན་གྱི་གདོན་ལས་ངེས་པར་ཐར་རོ། །སློབ་
དཔོན་ཡེ་ཤེས་རྡོ་རྗེས་མཛད་པ་རྫོགས་སོ།། ༄
