༄༅༅། །རྒྱ་གར་སྐད་དུ། བ་ལིཾ་བི་དྷི་ནཱ་མ། བོད་སྐད་དུ། གཏོར་མའི་ཆོ་ག་ཞེས་བྱ་བ། དཔལ་རྡོ་རྗེ་རྣམ་པར་འཇོམས་པ་ལ་ཕྱག་འཚལ་ལོ། །ཤ་ཆང་སྒོག་བཙོང་འབྱུང་པོའི་ཟས། །རྣམ་པར་སྤངས་པའི་ཁ་ཟས་ནི། །རིན་ཆེན་སྣོད་དུ་ལེགས་བཤམས་ཏེ། །ཁྲོ་བོའི་སྔགས་ཀྱིས་བགེགས་མེད་
བསང་། །སྟོང་པའི་སྔགས་ཀྱིས་རྣམ་སྦྱང་བ། །བྷྲཱུཾ་གྱུར་ས་ལས་རིན་ཆེན་སྣོད། །དེ་ཉིད་གསུམ་གྱིས་ཁ་དོག་དང་། །དྲི་རོ་ཕུན་སུམ་ཚོགས་པ་ལ། །ནམ་མཁའ་མཛོད་ཀྱིས་མངོན་སྤྲུལ་ལ། །བསམ་ཡས་མཆོད་པའི་སྤྲིན་དང་ལྡན། །རང་སྙིང་ས་བོན་འོད་ཟེར་གྱིས། །ཁྲོ་བོ་ཁྲོ་མོའི་འཁོར་ལོ་བསྐོར། །

སངས་རྒྱས་བྱང་ཆུབ་སེམས་དཔའ་དང་། །ཉན་ཐོས་རང་རྒྱལ་མཁའ་སྤྱོད་དང་། །ས་སྤྱོད་ས་འོག་སྤྱོད་པ་ཡི། །འགྲོ་བ་རྣམས་བཅས་མདུན་བལྟས་ཏེ། །མཆོད་བསྟོད་འབྱུང་པོ་ཀུན་སྔགས་དང་། །ཚིག་བརྗོད་ནས་ནི་ཚིམ་བྱས་ནས། །འདོད་དོན་སློང་དང་བསྐང་བ་དང་། །མུཿཡིག་སྔགས་
བཅས་གཤེགས་གསོལ་བྱ། །གཏོར་མའི་ཆོ་ག་རྫོགས་སོ།། ༄
