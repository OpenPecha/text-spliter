༄༅༅། །རྒྱ་གར་སྐད་དུ། སཾ་ཀྵིཔྟ་ཛཾ་བྷ་ལ་སཱ་དྷ་ནཾ། བོད་སྐད་དུ། ཛཾ་བྷ་ལའི་སྒྲུབ་ཐབས་མདོར་བསྡུས་པ། བཅོམ་ལྡན་འདས་ཛཾ་བྷ་ལ་ལ་ཕྱག་འཚལ་ལོ། །སྔོན་དུ་གསུངས་པའི་ཆོ་གས་སྙིང་གར་པདྨ་དང་། ཟླ་བ་ལ་ཛཾ་ཡིག་ལས་བྱུང་བའི་འོད་ཟེར་གྱིས་སངས་
རྒྱས་ལ་སོགས་པ་ཡང་དག་པར་མཆོད་ནས། སྡིག་པ་བཤགས་པ་ལ་སོགས་པ་བྱའོ། །དེ་ནས་སྟོང་པ་ཉིད་སྔོན་དུ་བྱ་སྟེ། དེ་ལ་དེ་བཞིན་དུ་གནས་པའི་སྣ་ཚོགས་པདྨ་དང་ཟླ་བ་ལ་ཛཾ་ལས་བྱུང་བའི་ས་བོན་གྱི་འོད་ཟེར་གྱིས་སེམས་ཅན་གྱི་དོན་མཛད་ནས། དེ་ཡོངས་སུ་གྱུར་པ་ལས་ཛཾ་བྷ་ལ་
གསེར་གྱི་མདོག་ལྟ་བུ་སངས་རྒྱས་ལྔའི་ཅོད་པན་ཅན། མཛེས་པའི་གོས་དང་། རྒྱན་གྱིས་སྤྲས་པ། ཞལ་གཅིག་ཕྱག་གཉིས་པ། ཕྱག་གཡས་ན་བཱི་ཛ་པཱུ་ར་ཀ་དང་། གཡོན་ན་ཁ་ནས་རིན་པོ་ཆེ་སྣ་ཚོགས་སྐྱུག་པའི་ནེའུ་ལེ་བསྣམས་པ། ཨུཏྤ་ལ་སྔོན་པོའི་ཕྲེང་བ་ཅན། ལོགས་
གཉིས་ན་རིན་པོ་ཆེའི་སྙེ་མའི་ཆར་འབེབས་པའི་གནོད་སྦྱིན་གཉིས་གནས་པ། ཞབས་ཀྱི་འོག་ན་གནས་པའི་རིན་པོ་ཆེའི་སྙེ་མའི་བུམ་པ་དཔག་ཏུ་མེད་པའི་འོད་ཟེར་གྱི་ཕྲེང་བ་འཁྲིགས་པའི་བདག་ཉིད་དུ་བསམ་པར་བྱའོ། །ཨོཾ་ཛཾ་བྷ་ལ་ཛ་ལེནྡྲ་ཡེ་སྭཱ་ཧཱ། ཞེས་པའི་སྔགས་བཟླས་སོ། །ལག་པ་གཉིས་ཁུ་
ཚུར་བཅིངས་ཏེ་གུང་མོའི་དབུས་སུ་ཚིགས་གསུམ་པ་བཅག་པའི་ཕྱག་རྒྱས་སྙིང་ག་དང་། མགྲིན་པ་དང་། སྤྱི་བོ་དང་མཛོད་སྤུ་རྣམས་སུ་སྦྱིན་པར་བྱའོ། །སྔ་བར་ཡང་དེ་བཞིན་དུ་སྐད་ཅིག་གིས་བསྒོམས་ནས། འདིས་ཆུ་སྙིམ་པ་བརྒྱ་རྩ་བརྒྱད་དབུལ་བར་བྱའོ། །ཨོཾ་ཛཾ་བྷ་ལ་ཛ་ལེནྡྲ་ཡེ་དྷ་ན་མེ་
དེ་ཧི་ད་ད་པཱ་ཡ་སྭཱ་ཧཱ། ཛཾ་བྷ་ལའི་སྒྲུབ་ཐབས་མདོར་བསྡུས་པ་རྫོགས་སོ།། ༄
