༄༅༅། །རྒྱ་གར་སྐད་དུ། བི་ཥེ་ཥ་ཨུ་པ་དེ་ཤཿ། བོད་སྐད་དུ། གདམས་ངག་གི་ཁྱད་པར་ཇི་ལྟ་བ་བཞིན་དུ་མདུད་པ་བྱས་པ་ཞེས་བྱ་བ་འདི་ལན་བདུན་དུ་སྟོད་གཡོགས་ཀྱི་གྲྭ་ལ་སྔགས་
མངོན་པར་ཟློ་བར་བྱེད་པ་སྟེ། ཕྱོགས་བཅུར་རྣམ་པར་བལྟས་ལ་མ་རུངས་པ་རྣམས་ཀྱི་ཁ་དང་། མིག་བདག་གིས་བཅིང་བར་བྱ་བ་ཞེས་བསམས་ཤིང་དབུགས་རྣམ་པར་བསྡམས་པས་ནང་དུ་བཟླས་པའི་རྣམ་པས་སྔགས་བརྗོད་པར་གྱུར་པས། །མདུད་པ་བཅིང་བར་བྱ་བ་ཡིན་ནོ། །གདམས་ངག་གི་
ཁྱད་པར་ཚུལ་ཁྲིམས་རྒྱལ་མཚན་གྱིས་བསྒྱུར་བའོ།། ༄
