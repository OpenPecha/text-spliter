༄༅༅། །རྒྱ་གར་སྐད་དུ། ཀུ་རུ་ཀུལླེ་སཱ་དྷ་ནཾ། བོད་སྐད་དུ། ཀུ་རུ་ཀུལླེའི་སྒྲུབ་ཐབས། ལྷ་མོ་ཀུ་རུ་ཀུལླེ་ལ་ཕྱག་འཚལ་ལོ། །ཆོ་ག་སྔར་བཤད་པ་དང་མཐུན་པ་ལས་གཞན་དུ་ན་ས་བོན་ཏཾ་དམར་པོ་ལས་བྱུང་བ། སྐུ་མདོག་དམར་པོ། པདྨ་དམར་པོ་དང་། ཟླ་བའི་གདན་ལ་བཞུགས་པ། རྒྱན་ཐམས་ཅད་ཀྱིས་
བརྒྱན་པ། ཕྱག་བཞི་པ་གཡས་མི་འཇིགས་པ། གཡོན་ཨུཏྤལ་དམར་པོ་བསྣམས་པ། ཡང་ཕྱག་གཉིས་ཀྱིས་ཨུཏྤ་ལ་དམར་པོ་ཁ་བྱེ་བ་ལྟ་བུའི་མདའ་དང་། གཞུ་རྣ་བའི་མཐར་ཐུག་པར་བཀང་སྟེ། བསྒྲུབ་པར་བྱ་བ་མདུན་དུ་བསམས་ལ་སྙིང་གར་བསྣུན་ཏེ། སྔར་བཤད་པའི་ཨུཏྤ་ལའི་
ཕྱག་རྒྱས་སྙིང་ག་དང་། དཔྲལ་བ་དང་། ལྐོག་མ་དང་། སྤྱི་བོར་གཏུགས་པར་བྱའོ། །དེ་ནས་བཟླས་བརྗོད་བྱ་སྟེ། ཨོཾ་ཀུ་རུ་ཀུལླེ་ཆེ་གེ་མོ་བ་ཤ་མཱ་ན་ཡ་སྭཱ་ཧཱ། དབང་དུ་བྱ་བའི་ཆོ་ག་རྫོགས་སོ།། །།བཟླས་པའི་དུས་སུ་ཁྱད་པར་དུ་འོད་ཟེར་གྱིས་སྡིག་སྒྲིབ་སྦྱངས་ཤིང་གཞན་དབང་དུ་བྱས་པར་ཡང་མོས་

པས་སོ། །དོན་ཡོད་གྲུབ་པས་རྒྱས་གདབ་བོ།། ༄
