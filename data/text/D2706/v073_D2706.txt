༄༅༅། །རྒྱ་གར་སྐད་དུ། བཱ་གཱི་ཤྭ་ར་སྟོ་ཏྲ། བོད་སྐད་དུ། ངག་གི་དབང་ཕྱུག་ལ་བསྟོད་པ། འཇམ་དཔལ་གཞོན་ནུར་གྱུར་པ་ལ་ཕྱག་འཚལ་ལོ། །
གཞོན་ནུའི་སྐུ་ལུས་འཆང་བ་པོ། །ཡེ་ཤེས་སྒྲོན་མས་རབ་བརྒྱན་ཅིང་། །འཇིག་རྟེན་གཏི་མུག་མུན་སེལ་བ། །འཇམ་པའི་དབྱངས་ལ་ཕྱག་འཚལ་བསྟོད། །ཉིད་ཀྱི་མདངས་བཟང་ཚོགས་ཀྱིས་ནི། །ཟླ་བ་རྒྱས་པ་ཟིལ་གནོན་ཅིང་། །རིག་པ་རྣམ་པར་གསལ་འགྱུར་བ། །འཇམ་དཔལ་
ཁྱེད་ལ་ཕྱག་འཚལ་བསྟོད། །མ་བགོས་ཟུར་ཕུད་གསལ་བ་ལ། །ནོར་བུ་འབར་བའི་སྐུ་མངའ་ཞིང་། །ཉིད་ནི་ཀུན་ཏུ་འབར་བའི་སྐུ། །འཇམ་དཔལ་ཁྱོད་ལ་ཕྱག་འཚལ་བསྟོད། །བདུད་རྩི་དྲི་ཞིམ་འཇམ་ལ་ཟབ། །ཡིད་འོང་གསུང་གི་སྒྲ་སྒྲོགས་པ། །དམ་པའི་ཆོས་ཀྱི་རིན་ཆེན་མཛོད། །འཇམ་
དཔལ་ཁྱོད་ལ་ཕྱག་འཚལ་བསྟོད། །གང་གཱའི་ཆུ་ཐིགས་ལྟར་གྲངས་མེད། །འཇིག་རྟེན་ཁམས་ཀུན་མ་ལུས་པ། །ཉོན་མོངས་དྲི་མ་རབ་སྤངས་པའི། །འཇམ་དཔལ་ཁྱོད་ལ་ཕྱག་འཚལ་བསྟོད། །མཆེ་བ་གཙིགས་ཤིང་བསྒྱིངས་པ་ཡི། །འཇིགས་པའི་སྐུ་ཡི་ཚུལ་གྱིས་ནི། །གཤིན་རྗེའི་
དྲེགས་པ་འཇོམས་མཛད་པ། །འཇམ་དཔལ་ཁྱོད་ལ་ཕྱག་འཚལ་བསྟོད། །རང་གི་སྤྱན་དང་མདངས་འདྲ་བའི། །ཨུཏྤལ་སྔོན་པོ་རབ་མཛེས་པ། །ལྷུང་བཟེད་ཕྱག་མཆོག་བསྣམས་པ་ཡི། །འཇམ་དཔལ་ཁྱོད་ལ་ཕྱག་འཚལ་བསྟོད། །ཕྱོགས་བཅུར་བཞུགས་པའི་སངས་རྒྱས་ལ། །
རྒྱ་ཆེར་བསྟོད་པ་སྤྱོད་པ་དང་། །འགྲོ་བའི་ལམ་ནི་སྟོན་པ་པོ། །འཇམ་དཔལ་ཁྱོད་ལ་ཕྱག་འཚལ་བསྟོད། །ངག་གི་དབང་ཕྱུག་ལ་བུ་མོ་བརྒྱད་ཀྱིས་བསྟོད་པ་རྫོགས་སོ།། ༄
