བཅོམ་ལྡན་འདས་དཔལ་ཕྱག་ན་རྡོ་རྗེ་ལ་ཕྱག་འཚལ་ལོ། །དུས་གསུམ་བདེ་གཤེགས་སྐུ་གསུང་སྲས་བཅས་ཀྱིས། །རྗེས་གནང་ལས་ཀྱི་ཁྲོ་བོར་
དབང་བསྐུར་བའི། །ཁྲོ་བོའི་རྒྱལ་པོ་དུག་དགྲ་དྲེགས་འཇོམས་ལ། །དང་བས་ཕྱག་འཚལ་སྒྲུབ་ཐབས་གོ་རིམས་བྲི། །གྲངས་དང་དུས་དང་མཚན་མ་ཡི། །བསྙེན་རྫོགས་དངོས་གྲུབ་བླང་བའི་ཕྱིར། །གནས་དང་ཕྱོགས་དང་སྔོན་འགྲོ་བྱ། །ཐིག་གདབ་ཚོན་དགྱེ་ཕྱག་མཚན་དགོད། །ཁྱད་
པར་རྫས་བསོག་བུམ་པ་བཅའ། །དངོས་གྲུབ་རྟེན་དང་མཆོད་རྫས་བཤམ། །སྡོམ་བསྲུང་སྨྱུང་གནས་ཆོས་གོས་བརྒྱན། །ཚོགས་བསགས་སྔོན་འགྲོ་རྗེས་འཇུག་དང་། །བསྐྱེད་རིམ་མཆོད་པ་བསྙེན་དུས་བཞིན། །བུམ་ནང་ལྷ་བསྐྱེད་སྤྱན་དྲངས་མཆོད། །དབང་བསྐུར་བྱིན་རླབས་བུམ་ནང་
ཐིམ། །གཟུངས་ཐག་བཏགས་ཏེ་རིག་སྔགས་བཟླས། །སྟེང་དང་འོག་གི་ལམ་གཉིས་ལས། །ཐུན་མོང་ཁྱད་པར་སྤྲོ་བསྡུ་བྱ། །ཁྲི་ཕྲག་བདུན་དུ་དངོས་གྲུབ་བླང་། །དབང་བསྐུར་མཆོད་བསྟོད་གཤེགས་སུ་གསོལ། །བསྐྱེད་བསྡུ་སྤྱོད་ལམ་རྣལ་འབྱོར་བྱ། །དེ་རྗེས་སྒྲུབ་པ་ཐོན་པ་དང་། །
ཞི་དང་བསྐྲད་དང་འཁྲུལ་འཁོར་དང་། །སྦྱོར་བ་བདུན་གསལ་ལས་ལ་སྦྱར། །སྒྲུབ་པའི་གོ་རིམས་སློབ་དཔོན་སུ་ཝཱ་གཱི་ཤྭ་ར་ཀཱིརྟིས་མཛད་པ་རྫོགས་སོ།། ༄
