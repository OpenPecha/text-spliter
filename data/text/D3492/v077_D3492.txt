༄༅༅། །རྒྱ་གར་སྐད་དུ། ཨཱརྱ་ཨཥྚ་མ་ཧཱ་བྷ་ཡ་ཏྲ་ཡ་ཏཱ་རཱ་ཡ་སཱ་དྷ་ནཾ། བོད་སྐད་དུ། འཕགས་པ་འཇིགས་པ་ཆེན་པོ་བརྒྱད་ལས་སྐྱོབ་པའི་སྒྲོལ་མའི་སྒྲུབ་ཐབས། །

འཕགས་མ་འཇིགས་པ་ཆེན་པོ་བརྒྱད་སྐྱོབ་མ་ལ་ཕྱག་འཚལ་ལོ། །གང་ཞིག་དྲན་པ་ཙམ་གྱིས་ནི། །འཇིགས་ཆེན་བརྒྱད་ནི་རབ་འཇོམས་མ། །དེ་ལ་ཕྱག་འཚལ་སྒྲུབ་པའི་ཐབས། །རབ་ཏུ་བཤད་བྱ་བྱས་རྣམས་ཉོན། །སྒྲོལ་མ་འཇིགས་པ་བརྒྱད་སྐྱོབ་མ། །ནམ་མཁའི་དབུས་སུ་
བསྒོམ་པར་བྱ། །ཡིད་ཀྱིས་རང་གིས་མཆོད་བྱས་ནས། །སྡིག་པ་རྣམས་ནི་བཤགས་པར་བྱ། །བསོད་ནམས་ལ་ནི་རྗེས་ཡིད་རང་། །ཕྱི་ནས་བདག་གི་བསོད་ནམས་བསག །དེ་ཕྱིར་བདག་ཉིད་དབུལ་བ་དང་། །གསུམ་ལ་ཡང་ནི་སྐྱབས་སུ་འགྲོ། །བྱ་བ་བྱས་པའི་ཕྱི་ནས་ནི། །གང་ལས་སྟོང་ཉིད་མཐའ་རུ་ནི། །
རང་གི་སྙིང་གར་ཟླ་དེའི་གནས། །པདྨ་འདབ་བརྒྱད་རྣམ་པར་རྒྱས། །དེ་ཡི་དབུས་སུ་ཏཱཾ་ཡིག་བསམ། །དེ་བཞིན་ཨུཏྤལ་སྔོན་པོ་ལ། །དེ་ཡི་དབུས་སུའང་ས་བོན་ལས། །མེ་ཡི་འོད་ཟེར་རབ་ཏུ་གསལ། །དེ་ཡིས་ལྷ་མོ་ཡང་དག་བསྐྱེད། །རྒྱན་རྣམས་ཀུན་གྱིས་རྣམ་པར་བརྒྱན། །ལྷ་མོའི་ཕྱག་གཡས་མཆོག་
སྦྱིན་དང་། །གཡོན་ན་དེ་བཞིན་ཨུཏྤལ་བསྣམས། །གཟུགས་མཛེས་ཞལ་ནི་གཅིག་མ་སྟེ། །ན་ཚོད་གཞོན་ནུས་ཡང་དག་གནས། །མེ་ཏོག་དྲི་བཟངས་དབུ་སྐྲ་བརྒྱན། །པདྨ་དང་ནི་ཟླ་བའི་སྟེང་། །སྐྱིལ་ཀྲུང་ཕྱེད་པའི་ཚུལ་གྱིས་བཞུགས། །ས་གསུམ་འཇིགས་པ་བརྒྱད་སྐྱོབ་པའི། །ལྷ་མོ་བརྒྱད་པོའི་
བར་དུ་ནི། །སྒྲོལ་མའི་སྐུ་རུ་བསྒོམ་པར་བྱ། །རྣལ་འབྱོར་པ་ནི་དེ་སྔགས་སྤྲོ། །ཡིད་ལ་ཞུ་བར་བསམ་པར་བྱ། །ཨོཾ་ཏཱ་རེ་ཏུཏྟཱ་རེ་ཏུ་རེ་སྭཱ་ཧཱ། །ཨོཾ་ཨོཾ་སྭཱ་ཧཱ་ཤར་དུའོ། །ཨོཾ་ཏཱ་སྭཱ་ཧཱ་ལྷོ་རུའོ། །ཨོཾ་རེ་སྭཱ་ཧཱ་ནུབ་ཏུའོ། །ཨོཾ་ཏུ་སྭཱ་ཧཱ་བྱང་དུའོ། །ཨོཾ་ཏྟཱ་སྭཱ་ཧཱ་མེའི་མཚམས་སུའོ། །ཨོཾ་རེ་སྭཱ་ཧཱ་བདེན་བྲལ་དུའོ། །ཨོཾ་ཏུ་སྭཱ་ཧཱ་རླུང་
དུའོ། །ཨོཾ་རེ་སྭཱ་ཧཱ་དབང་ལྡན་དུའོ། །དེ་ཐམས་ཅད་ཀྱང་འོད་ཟེར་འཕྲོ་བའི་གཟུགས་ཅན་ལྷ་མོ་ཇི་ལྟ་བ་དེ་བཞིན་དུ་གཞན་རྣམས་སོ། །ཨོཾ་ཏཱ་རེ་ཎྱཻ་བཛྲ་པུཥྤ་པྲ་ཏཱིཙྪ་སྭཱ་ཧཱ། ཞེས་པས་མེ་ཏོག་དབུལ་ཞིང་། དེ་བཞིན་དུ་དྲི་ལ་སོགས་པ་ཨུཏྤ་ལའི་ཕྱག་རྒྱས་དབུལ་བར་བྱའོ་ཞེས་སོ། །ལྷ་མོ་དགུ་དང་ལྡན་པ་ཡི། །དེ་བཞིན་
སྒྲུབ་ཐབས་འདི་བྱས་པས། །འཇིགས་པ་བརྒྱད་པོ་བཅོམ་གྱུར་ནས། །འགྲོ་ཀུན་བླ་མེད་བྱེད་པར་ཤོག །འཕགས་པ་འཇིགས་པ་ཆེན་པོ་བརྒྱད་ལས་སྐྱོབ་པའི་སྒྲོལ་མའི་སྒྲུབ་ཐབས་རྫོགས་སོ།། །།༄
