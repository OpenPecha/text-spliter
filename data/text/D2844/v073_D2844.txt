གཟིགས་མེའི་ལྷ་ལ་ཕྱག་འཚལ་ལོ། །དེ་ནས་མེ་ལྷའི་གཙོ་བོ་སྦྱིན་སྲེག་མཉེས། །ཞེས་འབྱུང་བའི་སྒྲུབ་ཐབས་བཤད་པར་བྱའོ། །འབར་བའི་དབུས་ན་བརྗིད་པའི་སྐུ། །མེ་ཐབ་དང་ནི་པདྨོ་འཛིན། །དམར་བའི་འོད་ཀྱིས་གདུག་པ་བསྲེག །སྒྲུབ་པོའི་དོན་མཛད་རྒྱལ་གྱུར་ཅིག །སྡིག་པ་
གཞིག་དང་དཔལ་བསྐྱེད་ཕྱིར། །ལས་ཀུན་སྒྲུབ་པར་བྱེད་པའི་རྒྱུ། །མེ་ཡི་ལས་ནི་ཕྱོགས་གཉིས་ལ། །མཁས་པས་འབད་དེ་རྣམ་པར་བརྟག །པདྨོ་ལྟ་བུའི་ཐབ་དཀར་མཛེས། །ཁྲུ་བཞི་ཁྲུ་གསུམ་ཁྲུ་གང་ཚད། །ཟླུམ་པོ་རྒྱ་ཕྱེད་སླར་བྲུས་པ། །སོར་བཞི་ཕྱིར་བསྐྱོད་བྱི་དོར་བྱུག །ཁ་ཁྱེར་སྟེགས་
ངོས་པདྨ་དང་། །རིན་ཆེན་དྲ་བ་དྲ་བ་ཕྱེད། །ཕྱི་རོལ་དཀྱིལ་འཁོར་འདབ་བརྒྱད་པ། །སྒོ་བཞིར་བཅས་དང་ཕོ་བྲང་ཉིད། །འོ་མ་འཛག་དང་དྲི་འབྱིན་དང་། །བཀྲ་ཤིས་བྱེད་པའི་ཡམ་ཤིང་ཚད། །ཅི་རིགས་དྲི་ཞིམ་སྔགས་བྱུགས་ལ། །བསངས་ནས་གཏོར་མ་སྔོན་འགྲོ་བས། །རོལ་མོ་

མཁན་དང་མེ་ཏོག་བརྒྱུད། །བྲམ་ཟེའི་ཁྱིམ་ནས་བླངས་པའམ། །ཡང་ན་རྡོ་རྗེ་ཡམ་ཤིང་ལ་སྩལ་མེ། །བསིལ་ཡབ་ཀྱིས་ནི་གཡབས་ཏེ་སྦར། །མར་དང་ཏིལ་དང་འབྲས་དང་ནས། །མེ་ཏོག་སྨན་དང་འབྲས་བུ་སྦྲང་། །ཞོ་ཟན་ཡུངས་ཀར་དཱུར་བ་ཤིང་། །མཐར་བྱེད་སྔགས་ཀྱིས་བསང་བར་བྱ། །ཨརྒ་
ལ་སོགས་བྱིན་བརླབས་ལ། །ལྷ་ཡི་སྦྱོར་བས་མེ་ལྷ་བསྐྱེད། །མེ་དང་ཡོ་བྱད་ཡམ་ཤིང་སྦྱངས། །ཡེ་ཤེས་བཀུག་པ་དབུལ་བར་བྱ། །ཨོཾ་བཛྲ་ཨགྣ་ཡེ་སྭཱ་ཧཱ། མར་དབུལ་བའོ། །ཨོཾ་བཛྲ་སརྦ་པཱ་པཾ་ད་ཧ་ན་བཛྲ་སྭཱ་ཧཱ། ཏིལ་དབུལ་བའོ། །ཨོཾ་བཛྲ་པུཥྚ་ཡེ་སྭཱ་ཧཱ། འབྲས་དབུལ་བའོ། །
ཨོཾ་བཛྲ་སན་ཏུཥྚ་ན་ཡེ་སྭཱ་ཧཱ། མེ་ཏོག་དབུལ་བའོ། །ཨོཾ་བཛྲ་ནིར་བྱཱ་དི་སྭཱ་ཧཱ། སྨན་དབུལ་བའོ། །ཨོཾ་བཛྲ་སརྦ་ཕ་ལ་ཡེ་སྭཱ་ཧཱ། འབྲས་བུ་དབུལ་བའོ། །ཨོཾ་བཛྲ་ཨ་ནུ་སྨྲྀ་ཏེ་སྭཱ་ཧཱ། སྦྲང་དབུལ་བའོ། །ཨོཾ་བཛྲ་སཾ་བ་དེ་སྭཱ་ཧཱ། ཞོ་ཟན་དབུལ་བའོ། །ཨོཾ་བཛྲ་ཨཱཿབིགྷྣཱན་སྭཱ་ཧཱ། བགེགས་བསལ་
བའོ། །ཨོཾ་བཛྲ་ཨཱ་ཡུ་ཥེ་སྭཱ་ཧཱ། དཱུར་བ་དབུལ་བའོ། །ཨོཾ་བཛྲ་བོ་དྷི་བྲྀཀྵ་ཡེ་སྭཱ་ཧཱ། ཡམ་ཤིང་དབུལ་བའོ། །ཨོཾ་ཨཱཿཧྲཱིཿན་མཿས་མནྟ་བུདྡྷཱ་ནཱཾ་ཨོཾ་ཨགྣ་ཡེ་ཨ་མུ་ཀ་སྱ་སརྦ་ཀརྨ་སིདྡྷི་ཀུ་རུ་སྭཱ་ཧཱ། བསྲེག་རྫས་ཐམས་ཅད་ཀྱི་མགོའམ་མཇུག་དུ་དགང་བླུགས་དབུལ་བའོ། །གཞན་ཡང་
མཆོད་པ་ཐམས་ཅད་ཀྱང་། །ཅི་རིགས་མཁས་པས་ཤེས་བྱས་ཏེ། །དགང་དང་བླུགས་པའི་ཕྱག་རྒྱ་ཡིས། །མཉེས་བྱས་ཁྲུས་དང་བཅས་པས་མཆོད། །མཐར་བཟོད་པར་གསོལ་བ་དངོས་གྲུབ་བླངས་ལ་གཤེགས་སུ་གསོལ་ནས་བདེ་བ་ལ་རེག་པར་གནས་པ་བྱའོ། །སྦྱིན་སྲེག་གི་ཆོ་ག་
