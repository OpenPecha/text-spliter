ཀྱང་བྱིན་གྱིས་བརླབས་ལ། སྙིང་གའི་ས་བོན་ལས་ངེས་པར་གྲུབ་པའི་ཨུཏྤལ་དམར་པོ་ལས་སྐྱེས་པའི་ཟླ་བའི་དཀྱིལ་འཁོར་ལ་ས་བོན་དེ་ཉིད་བསྒོམ་པར་བྱ་བའོ། །དེ་དག་ཐམས་ཅད་ཡོངས་སུ་གྱུར་པ་ལས་བཅོམ་ལྡན་འདས་མ་སྟེ། སྐུ་མདོག་དམར་མོ་པདྨ་དང་ཟླ་བ་དམར་པོའི་
གདན་ལ་ན་བཟའ་དམར་པོ་དང་དབུ་རྒྱན་དང་ལྡན་པ། ཕྱག་བཞི་གཡས་པས་མི་འཇིགས་པ་རབ་ཏུ་སྦྱིན་པ། གཞན་གྱིས་ནི་ལེགས་པར་བཀང་བའི་མདའོ། །གཡོན་པའི་གཅིག་གིས་ནི་རིན་པོ་ཆེའི་གཞུ་བསྣམས་པ། གཞན་གྱིས་ནི་ཨུཏྤ་ལ་དམར་པོ་བསྣམས་པ་དབུ་རྒྱན་
ལ་སྣང་བ་མཐའ་ཡས་ཀུ་རུ་ཀུལླེའི་ཁོང་ན་བཞུགས་པར་གྱུར་པའི་སྒེག་པའི་ཉམས་དང་ལྡན་ཞིང་མཆོག་ཏུ་འབར་བར་གྱུར་པའི་གཞོན་ནུ་ལང་ཚོ་ལ་བབ་པའི་ཀུ་རུ་ཀུལླེ་ཇི་སྲིད་དུ་སྐྱོ་བར་མ་གྱུར་པ་དེ་སྲིད་དུ་བསྒོམ་པར་བྱ་བའོ། །སྐྱོ་བར་ནི་སྔགས་བཟླས་པར་བྱ་བའོ། །དེ་ལ་

བསྒྲུབ་བྱ་སྐྲ་གྲོལ་བ་གཅེར་བུར་གྱུར་ཅིང་། མྱོས་པ་རྡོག་པས་མནན་ཏེ། སྙིང་ཨུཏྤལ་དམར་པོའི་མདས་འབིགས་བཞིན་པས་སྔགས་བཟླས་པར་བྱ་བའོ། །དེ་ལ་སྔགས་ནི་འདི་ཡིན་ཏེ། ཨོཾ་ཀུ་རུ་ཀུལླེ་ཧྲཱིཿཆེ་གེ་མོ་བདག་གི་དབང་དུ་གྱིས་སྭཱ་ཧཱ་ཞེས་བྱ་བའོ། །ཀུ་རུ་ཀུལླེའི་
