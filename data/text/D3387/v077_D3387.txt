༄༅༅། །རྒྱ་གར་སྐད་དུ། བཛྲ་སྲྀཾ་ཁ་ལཱ་སཱ་དྷ་ནཾ། བོད་སྐད་དུ། རྡོ་རྗེ་ལྕགས་སྒྲོག་གི་སྒྲུབ་པའི་ཐབས། རྡོ་རྗེ་ལྕགས་

སྒྲོག་མ་ལ་ཕྱག་འཚལ་ལོ། །སྔོན་དུ་བསྟན་པའི་ཆོ་གས་རང་གི་སྙིང་གར་ཟླ་བའི་དཀྱིལ་འཁོར་ལ་ཧཱུཾ་ལྗང་གུའི་ཡི་གེ་ལས་ངེས་པར་རྫོགས་པའི་རྡོ་རྗེ་ལྕགས་སྒྲོག་མ་ཞལ་གསུམ་ཕྱག་དྲུག་མ། སྔོན་པོ་དང་དཀར་པོ་ནི་གཡས་དང་། ཅིག་ཤོས་ཀྱི་ཞལ་ལོ། །ཞལ་རེ་རེ་ལ་སྤྱན་གསུམ་གསུམ་མོ། །རྡོ་རྗེ་དང་རྡོ་རྗེ་
ལྕགས་སྒྲོག་དང་། མདའ་ནི་གཡས་ཀྱི་ཕྱག་གསུམ་ནའོ། །སྡིགས་མཛུབ་དང་། ཞགས་པ་དང་། གཞུ་ནི་གཡོན་གྱི་ཕྱག་གསུམ་ནའོ། །མདོག་ལྗང་སེར་ཞལ་ཅུང་ཞིག་བཞད་པ། རྒྱན་ཐམས་ཅད་ཀྱིས་བརྒྱན་པ། གདུག་པ་ཅན་གྱི་སེམས་ཅན་འདུལ་བའི་བདག་ཉིད་དུ་བསམས་ལ། ཕྱག་རྒྱ་
བཅིང་བར་བྱ་སྟེ། དེ་ལ་ལག་པ་གཉིས་སོ་སོར་ཁུ་ཚུར་བཅིངས་ནས། མཐེའུ་ཆུང་དང་སྡིགས་མཛུབ་གཉིས་ལྕགས་སྒྲོག་གི་རྣམ་པ་ཅན་དུ་བཅིང་བར་བྱ་བ་ནི་རྡོ་རྗེ་ལྕགས་སྒྲོག་གི་ཕྱག་རྒྱའོ། །དེ་ནས་བཟླས་པ་ནི། ཨོཾ་བཛྲ་སྲིཾ་ཁ་ལེ་ཧཱུཾ་ཕཊ་སྭཱ་ཧཱ་ཞེས་བྱའོ། །རྡོ་རྗེ་ལྕགས་སྒྲོག་གི་སྒྲུབ་ཐབས་རྫོགས་སོ། །
པཎྜི་ཏ་དོན་ཡོད་རྡོ་རྗེ་དང་། ཁམས་པ་ལོ་ཙཱ་བ་དགེ་སློང་བ་རིས་བསྒྱུར་བའོ།། ༄
