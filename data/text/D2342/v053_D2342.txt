༄༅༅། །རྒྱ་གར་སྐད་དུ། ཏཏྟྭཱ་སྭ་བྷཱ་བ་དྲྀཥྚི་གཱི་ཏི་ཀ་དོ་ཧ་ནཱ་མ། བོད་སྐད་དུ། རང་བཞིན་གྱི་དོ་ཧ་མཛོད་ཀྱི་
གླུའི་ལྟ་བ་ཞེས་བྱ་བ། དཔལ་ཧེ་རུ་ཀ་ལ་ཕྱག་འཚལ་ལོ། །གཉིས་མེད་བདུད་རྩི་ཀུན་འདུས་གྲུབ་ཐོབ་ཀྱི། །དྲི་མེད་དགོངས་པ་ལྟ་སྤྱོད་ལ་སོགས་ནི། །ཉམས་ལེན་ཐབས་དེ་ཀུན་ལ་ཕན་པའི་ཕྱིར། །བདེ་གཤེགས་
རྣམས་ལ་ཕྱག་འཚལ་བཤད་པར་བྱ། །དངོས་པོ་ཡང་ནི་མ་ཡིན་ཞིང་། །དངོས་མེད་ལ་ནི་མ་སྐྱེས་པའི། །འདི་ལྟ་བུ་ཡི་རང་བཞིན་ནི། །སུ་ཞིག་གིས་ནི་རྟོགས་པར་འགྱུར། །ལཱུ་ཡི་པ་ནི་སྨྲ་བར་བྱེད། །འདི་ལྟ་བུ་
ནི་ངོ་མཚར་ཆེ། །ཁམས་གསུམ་གནས་སུ་ལོངས་སྤྱོད་ཀྱང་། །དེ་ཉིད་གནས་ནི་རྟོགས་པ་མེད། །དེ་ཡི་ཁ་དོག་རིགས་དང་ནི། །དབྱིབས་ཀྱང་བདག་གིས་རྟོགས་པ་མེད། །དེ་ལྟ་བུ་ཡི་རང་བཞིན་ནི། །དངོས་པོར་མ་
གྲུབ་ཇི་ལྟར་བསྟན། །ལཱུ་ཡི་པ་ནི་སྨྲ་བར་བྱེད། །འདི་ལྟ་བུ་ནི་ངོ་མཚར་ཆེ། །ཁམས་གསུམ་གནས་སུ་ལོངས་སྤྱོད་ཀྱང་། །དེ་ཉིད་གནས་ནི་རྟོགས་པ་མེད། །བདག་གིས་གཞན་ལ་ཇི་ལྟར་དྲི། །ཆུ་གཏིང་ཟླ་བའི་གཟུགས་

བརྙན་ནི། །དོན་དམ་སྒྱུ་མར་རྟོགས་པ་མེད། །ལཱུ་ཡི་པ་ནི་སྨྲ་བར་བྱེད། །འདི་ལྟ་བུ་ནི་ངོ་མཚར་ཆེ། །ཁམས་གསུམ་གནས་སུ་ལོངས་སྤྱོད་ཀྱང་། །དེ་ཉིད་གནས་ནི་རྟོགས་པ་མེད། །ལཱུ་ཡི་པ་ནི་སྨྲ་བར་བྱེད། །བདག་གིས་གཞན་ལ་ཇི་ལྟར་དྲི། །དེ་ཡི་རང་བཞིན་རྟོགས་པ་
