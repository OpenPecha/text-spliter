༄༅༅། །གང་ཡང་མི་སྐྱེ་མི་འགག་དངོས་པོ་བཅིང་བ་མེད། །ཁམས་གསུམ་གྲོལ་མེད་འཁྲུལ་པ་རྣམས་ལ་སྣང་། །རྩ་བ་སེམས་ཉིད་སྣ་ཚོགས་འཁྲུལ་པའི་རང་བཞིན་ནི། །ཐོག་མེད་འཁྲུལ་པའི་རང་བཞིན་གཞིགས་ན་
རྙེད་མི་འགྱུར། །དངོས་རྣམས་གདོད་ནས་སྐྱེ་མེད་མཐོང་བ་གང་ཞིག་ཡིན། །སུ་ཞིག་སྐྱེ་བར་འགྱུར་དང་འགག་དང་བཅིང་བ་སེམས། །རྩ་བ་སེམས་ཉིད་སྣ་ཚོགས་འཁྲུལ་པའི་རང་བཞིན་ནི། །ཐོག་མེད་འཁྲུལ་པའི་རང་བཞིན་གཞིགས་ན་རྙེད་མི་འགྱུར། །སེམས་ནི་ནམ་མཁའ་དྲི་མེད་
དང་འདྲ་སློབ་དཔོན་གསུང་། །སྲིད་པ་གསུམ་པོ་དེ་འདྲའི་རང་བཞིན་ཡིན། །རྩ་བ་སེམས་ཉིད་སྣ་ཚོགས་འཁྲུལ་པའི་རང་བཞིན་ནི། །ཐོག་མེད་འཁྲུལ་པའི་རང་བཞིན་གཞིགས་ན་རྙེད་མི་འགྱུར། །རང་བཞིན་མི་རྟོག་རང་རིག་ཉམས་སུ་མྱོང་བས་འགྲུབ། །རང་བཞིན་འོད་གསལ་འཁྲུལ་པས་
ཅི་ཕྱིར་རྨོངས་པར་འགྱུར། །རྩ་བ་སེམས་ཉིད་སྣ་ཚོགས་འཁྲུལ་པའི་རང་བཞིན་ནི། །ཐོག་མེད་འཁྲུལ་པའི་རང་བཞིན་གཞིགས་ན་རྙེད་མི་འགྱུར། །བླ་མ་མཻ་ཏྲིའི་གླུ་རྫོགས་སོ།། ༄
